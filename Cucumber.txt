WITH TopStudents AS (
    SELECT 
        student_id,
        name,
        math_marks,
        ROW_NUMBER() OVER (ORDER BY math_marks DESC) AS row_num
    FROM 
        students
)
SELECT 
    (SELECT SUM(math_marks) FROM TopStudents WHERE row_num <= 5) AS total_marks,
    student_id,
    name,
    math_marks
FROM 
    TopStudents
WHERE 
    row_num <= 5;


-- Create the students table
CREATE TABLE students (
    student_id INT PRIMARY KEY,
    name VARCHAR(50),
    math_marks INT
);
