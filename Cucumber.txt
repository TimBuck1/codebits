WITH TopStudents AS (
    SELECT 
        student_id,
        name,
        math_marks,
        ROW_NUMBER() OVER (ORDER BY math_marks DESC) AS row_num
    FROM 
        students
)
SELECT 
    (SELECT SUM(math_marks) FROM TopStudents WHERE row_num <= 5) AS total_marks,
    student_id,
    name,
    math_marks
FROM 
    TopStudents
WHERE 
    row_num <= 5;
