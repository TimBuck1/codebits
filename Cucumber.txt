-- Drop the table if it already exists
BEGIN
    EXECUTE IMMEDIATE 'DROP TABLE students';
EXCEPTION
    WHEN OTHERS THEN
        IF SQLCODE != -942 THEN
            RAISE;
        END IF;
END;
/

-- Create the students table
CREATE TABLE students (
    student_id NUMBER PRIMARY KEY,
    name VARCHAR2(50),
    math_marks NUMBER
);

-- Query to get the sum of marks for top 5 students and their details
WITH TopStudents AS (
    SELECT 
        student_id,
        name,
        math_marks
    FROM (
        SELECT 
            student_id,
            name,
            math_marks,
            ROW_NUMBER() OVER (ORDER BY math_marks DESC) AS row_num
        FROM 
            students
    )
    WHERE row_num <= 5
)
SELECT 
    student_id,
    name,
    math_marks,
    NULL AS total_marks
FROM 
    TopStudents

UNION ALL

SELECT 
    NULL AS student_id,
    'Total' AS name,
    NULL AS math_marks,
    SUM(math_marks) AS total_marks
FROM 
    TopStudents;
